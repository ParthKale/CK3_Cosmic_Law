# Day-based transit: add 91 days (quarter) and advance planets when days >= transit_days.
# Only when Cosmic Law is not Off.
cosmic_law_quarterly_transit_tick = {
	if = {
		limit = { NOT = { has_game_rule = cosmic_law_scope_off } }
		change_global_variable = { name = global_surya_days add = 91 }
		change_global_variable = { name = global_chandra_days add = 91 }
		change_global_variable = { name = global_mangal_days add = 91 }
		change_global_variable = { name = global_budh_days add = 91 }
		change_global_variable = { name = global_guru_days add = 91 }
		change_global_variable = { name = global_shukra_days add = 91 }
		change_global_variable = { name = global_shani_days add = 91 }
		change_global_variable = { name = global_rahu_days add = 91 }
		change_global_variable = { name = global_ketu_days add = 91 }
		claw_advance_surya_tick = yes
		claw_advance_chandra_tick = yes
		claw_advance_mangal_tick = yes
		claw_advance_budh_tick = yes
		claw_advance_guru_tick = yes
		claw_advance_shukra_tick = yes
		claw_advance_shani_tick = yes
		claw_advance_rahu_tick = yes
		claw_advance_ketu_tick = yes
	}
}

# Helper: advance one planet one step (rashi 12->1 else +1, days -= transit). Used in repeat blocks.
claw_advance_surya_one = {
	if = {
		limit = { global_var:global_surya_days >= 30 }
		if = { limit = { global_var:global_surya_rashi == 12 } set_global_variable = { name = global_surya_rashi value = 1 } }
		else = { change_global_variable = { name = global_surya_rashi add = 1 } }
		change_global_variable = { name = global_surya_days add = -30 }
		claw_debug_toast_surya = yes
	}
}
claw_advance_surya_tick = {
	claw_advance_surya_one = yes
	claw_advance_surya_one = yes
	claw_advance_surya_one = yes
	claw_advance_surya_one = yes
}
